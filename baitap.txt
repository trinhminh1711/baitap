USE roomchat;
 CREATE TABLE ROOM
 (
     IDROOM CHARACTER PRIMARY KEY,
     IDtaikhoan CHARACTER,
     IDFRI CHARACTER, 
     IDMESS CHARACTER
   
);
CREATE TABLE taikhoan 
(
   IDtaikhoan CHARACTER  PRIMARY KEY ,
   TEN CHARACTER ,
    TUOI INT,
    GIOITINH CHARACTER,
     FOREIGN KEY (IDtaikhoan) REFERENCES ROOM (IDROOM)
   
);
  CREATE TABLE tinnhan 
(
      IDMESS CHARACTER  PRIMARY KEY ,
      IDtaikhoan CHARACTER ,
      IDSEND CHARACTER,
      IDHEARD CHARACTER ,
      NOIDUNG TEXT,
   
      FOREIGN KEY (IDtaikhoan) REFERENCES taikhoan (IDtaikhoan),
     FOREIGN KEY (IDMESS) REFERENCES ROOM (IDROOM)
);
CREATE TABLE FRIEND
( 
    IDADD CHARACTER PRIMARY KEY ,
    IDFRI CHARACTER,
    IDtaikhoan CHARACTER,
    FOREIGN KEY (IDtaikhoan) REFERENCES taikhoan (IDtaikhoan),
     FOREIGN KEY (IDFRI) REFERENCES ROOM (IDROOM)
    
 );
